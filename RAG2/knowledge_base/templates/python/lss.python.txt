def _kph_to_mps(kph: float) -> float:
    return kph / 3.6

    def _initialize_actors(self, config):
        """
        Spawn/position actors.
        - Ego is typically already spawned by ScenarioRunner; still safe to set transform if required.
        - Target vehicle is spawned here as "other actor".
        """

        ego = self.ego_vehicles[0]

        # Optionally set ego to a known transform (if your pipeline needs it)
        ego.set_transform(self.ego_spawn_transform)

        # Spawn target
        bp_lib = self.world.get_blueprint_library()
        tgt_bp = bp_lib.find(self.target_blueprint)

        target_actor = self.world.try_spawn_actor(tgt_bp, self.target_spawn_transform)
        if target_actor is None:
            raise RuntimeError("Failed to spawn target actor with blueprint: {{ target_blueprint }}")

        self.other_actors.append(target_actor)

    def _create_behavior(self):
        """
        Behavior tree:
        1) Place actors
        2) Wait until trigger (distance or TTC)
        3) Ego drives at constant speed
        4) Target drives, then performs lane change (LaneChange atomic)
        5) Run for post-trigger duration
        """

        ego = self.ego_vehicles[0]
        target = self.other_actors[0]

        ego_speed_mps = _kph_to_mps(self.ego_speed_kph)
        target_speed_mps = _kph_to_mps(self.target_speed_kph)

        root = py_trees.composites.Sequence(name="LSS_Root")

        # 1) Ensure transforms are set at start
        root.add_child(ActorTransformSetter(ego, self.ego_spawn_transform))
        root.add_child(ActorTransformSetter(target, self.target_spawn_transform))

        # 2) Trigger condition: prefer TTC if provided, else use distance
        # NOTE: LLM should pick one by filling the other as None/null.
        if self.ttc_trigger_s is not None:
            trigger = InTimeToArrivalToVehicle(
                other_actor=target,
                ego_vehicle=ego,
                time=float(self.ttc_trigger_s)
            )
        else:
            trigger = InTriggerDistanceToVehicle(
                other_actor=target,
                ego_vehicle=ego,
                distance=float(self.trigger_distance_m)
            )
        root.add_child(trigger)

        # 3) Ego motion
        root.add_child(WaypointFollower(ego, ego_speed_mps))

        # 4) Target motion (before lane change)
        root.add_child(WaypointFollower(target, target_speed_mps))

        # 4b) Lane change (target)
        # LaneChange atomic is used here as a placeholder behavior.
        # The exact parameters vary across ScenarioRunner versions; keep these placeholders fixed.
        # If your ScenarioRunner build uses a different signature, update template v2 later.
        if self.lane_change_direction.lower() == "left":
            direction = "left"
        else:
            direction = "right"

        root.add_child(
            LaneChange(
                target,
                direction=direction,
                distance=self.lane_change_distance_m,
                duration=self.lane_change_duration_s
            )
        )

        # 5) Let scenario run for a fixed duration after lane change
        root.add_child(Idle(duration=float({{ post_trigger_duration_s }})))

        return root
