def _kph_to_mps(kph: float) -> float:
    return kph / 3.6


class {{ scenario_class_name }}(BasicScenario):
    """

    VRU scenario template intended for cases like:
    - pedestrian crossing
    - bicyclist crossing
    - VRU entering lane / crossing path of ego

    def __init__(self, world, ego_vehicles, config, randomize=False, debug_mode=False, criteria_enable=True):

        # ----------------------------

        self.vru_blueprint = "{{ vru_blueprint }}"   # e.g., "walker.pedestrian.0001" OR "vehicle.bh.crossbike"

        # Speeds (kph)
        self.ego_speed_kph = float({{ ego_speed_kph }})
        self.vru_speed_kph = float({{ vru_speed_kph }})

    def _initialize_actors(self, config):
        """
        Spawn/position actors.
        - Ego is typically already spawned by ScenarioRunner; still safe to set transform if required.
        - VRU is spawned here as "other actor".
        """

        # Ensure ego exists
        ego = self.ego_vehicles[0]

        # Optionally set ego to a known transform (if your pipeline needs it)
        ego.set_transform(self.ego_spawn_transform)

        # Spawn VRU
        bp_lib = self.world.get_blueprint_library()
        vru_bp = bp_lib.find(self.vru_blueprint)

        vru_actor = self.world.try_spawn_actor(vru_bp, self.vru_spawn_transform)
        if vru_actor is None:
            raise RuntimeError("Failed to spawn VRU actor with blueprint: {{ vru_blueprint }}")

        self.other_actors.append(vru_actor)

    def _create_behavior(self):
        """
        Behavior tree:
        1) Place actors
        2) Wait until trigger (distance or TTC)
        3) Move VRU toward target
        4) Keep ego driving at constant speed (handled via WaypointFollower on ego OR external controller)
        5) End
        """

        ego = self.ego_vehicles[0]
        vru = self.other_actors[0]

        ego_speed_mps = _kph_to_mps(self.ego_speed_kph)
        vru_speed_mps = _kph_to_mps(self.vru_speed_kph)

        root = py_trees.composites.Sequence(name="VRU_Root")

        # 1) Ensure transforms are set at start
        root.add_child(ActorTransformSetter(ego, self.ego_spawn_transform))
        root.add_child(ActorTransformSetter(vru, self.vru_spawn_transform))

        # 2) Trigger condition: prefer TTC if provided, else use distance
        # NOTE: LLM should pick one by filling the other as None/null.
        if self.ttc_trigger_s is not None:
            trigger = InTimeToArrivalToVehicle(
                other_actor=vru,
                ego_vehicle=ego,
                time=float(self.ttc_trigger_s)
            )
        else:
            trigger = InTriggerDistanceToVehicle(
                other_actor=vru,
                ego_vehicle=ego,
                distance=float(self.trigger_distance_m)
            )

        root.add_child(trigger)

        # 3) Ego motion (simple follower; if your stack uses external control, LLM may set speed low or use Idle)
        # WaypointFollower expects the actor to be a vehicle. This is for ego only.
        root.add_child(WaypointFollower(ego, ego_speed_mps))

        # 4) VRU motion:
        # If VRU is a vehicle (bicycle), we can use WaypointFollower.
        # If VRU is a walker, the LLM should keep vru_blueprint consistent with vehicle/walker support.
        root.add_child(WaypointFollower(vru, vru_speed_mps))

        # 5) Let scenario run for a fixed duration after trigger to complete interaction
        root.add_child(Idle(duration=float({{ post_trigger_duration_s }})))

        return root

    def _create_test_criteria(self):
        """
        Minimal evaluation criteria:
        - Collision test between ego and other actors
        You can add more criteria later via atoms.
        """
        criteria = []
        if self.criteria_enable:
            criteria.append(CollisionTest(self.ego_vehicles[0]))
        return criteria
